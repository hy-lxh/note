import * as THREE from "three.js";let canvas = document.querySelector("canvas"),context = canvas.getContext("webgl2",{alpha: false}); let width = canvas.offsetWidth,height = canvas.offsetHeight;let scene = new THREE.Scene();//创建一个渲染场景let camera = new THREE.PerspectiveCamera(75,width / height,1,400);//创建一个相机 参数分别是角度、渲染区域宽高比、近截面、远截面let renderer = new THREE.WebGLRenderer();//里面可以设置渲染在哪个canvas,比如下面这个options,创建一个渲染器//{canvas,context}document.body.appendChild(renderer.domElement);//将渲染完成的canvas元素添加到body里 let geometry = new THREE.BoxGeometry(1,1,1);//创建一个正方体let material = new THREE.MeshBasicMaterial({ color: "#ff00f0" });//创建一个材质let cube = new THREE.Mesh(geometry,material);//将材质贴到正方体上 scene.add(cube);//将正方体加入到场景中,添加元素都是在(0,0,0)点 //camera.position.z = 5;// 所以需要移动相机的位置，不然就和正方体重合在一起了，还可以直接通过camera.position.set(x,y,z);camera.position.set(1,1.2,5); renderer.render(scene,camera);//将相机和场景都添加到渲染器里面，开始渲染

import * as THREE from "three.js";let width = window.innerWidth,height = window.innerHeight;let scene = new THREE.Scene();let camera = new THREE.PerspectiveCamera(75,width / height,1,500); let renderer = new THREE.WebGLRenderer();document.body.appendChild(renderer.domElement);renderer.setSize(width,height);let geometry = new THREE.Geometry();geometry.vertices.push(new THREE.Vector3(-10,0,0));geometry.vertices.push(new THREE.Vector3(0,10,0));geometry.vertices.push(new THREE.Vector3(10,0,0)); let material = new THREE.LineBasicMaterial({ color: "#00ff00" });let line = new THREE.Line(geometry,material); let geometry1 = new THREE.Geometry();geometry1.vertices = [  new THREE.Vector3(-10,5,0),  new THREE.Vector3(0,15,0),  new THREE.Vector3(10,5,0)];let line1 = new THREE.Line(geometry1,material);scene.add(line1);scene.add(line);camera.position.set(0,0,100);camera.lookAt(0,0,0); renderer.render(scene,camera);

材质

MeshBasicMaterial基础材质，不受关照影响
MeshLambertMaterial与光照有反应，漫反射
MeshPhongMaterial高光材质
MeshStandardMaterial相比较Phong材质，这个材质对玻璃和金属材料更加真实

使用

Basic和Lambert

Basic和Lambert两种材质属性是一样的,就是Lambert受光照影响,Basic不受影响
