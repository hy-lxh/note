class Scope{  this.$$watchers = [];}///**//* name: 变量名//* value: 变量新值//* listener: 更新函数//**/Scope.prototype.$watch = function(name,value,listener){  this.$$watchers.push({    name,    value,    last: '',//旧值    listener: listener || function(){}  });}//调用更新函数以及value Scope.prototype.$digest = function(){  let digest = true,bindList = document.querySelectAll("[ng-bind]"),name;  while(digest){    digest = false;    for(let watcher of this.$$watchers){      let newVal = watcher.value,oldVal = watcher.last;      if(newVal !== oldVal && !isNaN(newVal)){        digest = true;        watcher.listener(newVal,oldVal);        for(let i = 0;i < bindList.length;i++){          name = bindList[i].getAttribute("ng-bind");          if(bindList[i].tagName === "INPUT"){            bindList[i].value = this[name];          }else{            bindList[i].innerHTML = this[name];          }        }        watcher.last = newVal;      }    }  }}
