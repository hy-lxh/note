import {useReducer} from "react";function reducer(state,action){  switch(action.type){    case '+': return { count: state.count + 1 };    case '-': return { count: state.count - 1 };    default: new Error('')  }}function MyComponent(){  const initialState = {    count: 1  }  const [state,dispatch] = useReducer(reducer,initialState);  return (    <div>      <button onClick={ () => dispatch({type: '+' }) }>+</button>      <button onClick={ () => dispatch({ type: '-' }) }></button>    </div>  );}
