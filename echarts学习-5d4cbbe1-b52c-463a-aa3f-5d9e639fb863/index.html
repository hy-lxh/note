let chart = excharts.init(dom,主题);const options = {  title: {    text: "学习echarts",//主标题    subtext: "基础折线图"//副标题  },  aAxis: {    type: "category",//x轴类型    data: ["A","B","C"]//x轴数据  },  yAixs: {    type: "value"//y轴类型,就是值  },  series: [//存放数据    {      type: "line",//画图的类型      data: [10,"-",9],//画图的数据,空数据可以用-隔开      lineStyle: {//设置折线图样式        color: "pink",//线的颜色        width: 10,//线的宽度        type: "dashed",//线的类型        cat: "butt",//线的交界处类型,有butt,square,round        shadowColor: "#000",//设置阴影颜色,        shadowBlur: 10,//设置值        shadowOffsetX: 10,//设置x方向阴影偏移量        shadowOffsetY: 9,//y轴阴影量        opacity: 0.7  //透明度0 - 1      },      label: {//显示数值        show: true,        position: "bottom",//位置        textStyle: {//文字样式          color: "yellow",//文字颜色          fontSize: 14,//文字大小          fontWeight: 600//文字粗细        }      }    }  ]};chart.setOption(options);
