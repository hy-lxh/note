Function.prototype.before = function(fn){  const _target = this;//这里面this就是指链式调用的那个this  return function(){    fn.apply(this,arguments);    return _target.apply(this,arguments);  }}Function.prototype.after = function(afterFn){  let _target = this;  return function(){    let ret = _target.apply(this,arguments);    afterFn.apply(this.arguments);    return ret;  }}
